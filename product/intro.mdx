import React, { useState } from 'react';

// 定义 B站视频ID
const bvid = "BV1xx411c7mD";  // 改成你的B站视频ID

// 视频播放器组件
const BilibiliPlayer = () => {
  const [isPlaying, setIsPlaying] = useState(false);
  
  const videoUrl = `https://player.bilibili.com/player.html?bvid=${bvid}&autoplay=${isPlaying}&high_quality=1`;

  return (
    <div style={{ marginTop: '20px' }}>
      {/* 播放控制按钮 */}
      <button 
        onClick={() => setIsPlaying(!isPlaying)}
        style={{
          padding: '10px 20px',
          marginBottom: '15px',
          backgroundColor: isPlaying ? '#ff4d4d' : '#00a1d6',
          color: 'white',
          border: 'none',
          borderRadius: '4px',
          cursor: 'pointer'
        }}
      >
        {isPlaying ? '暂停视频' : '播放视频'}
      </button>

      {/* B站播放器 */}
      <iframe 
        src={videoUrl}
        width="100%" 
        height="500px"
        frameBorder="no"
        scrolling="no"
        allowFullScreen={true}
        style={{
          borderRadius: '8px',
          boxShadow: '0 2px 8px rgba(0,0,0,0.1)'
        }}
      />
    </div>
  );
};

{/* 在文档中使用播放器 */}
